ETL Shell commands

cut

echo "database" | cut -c1-4
echo "database" | cut -c5-8   --> from 5 to 8
echo "database" | cut -c1,5   --> cut char 1 and 5

extract fields:

cut -d":" -f1 /etc/passwd   --> field 1
cut -d":" -f1,3,6 /etc/passwd   --> field 1,3 and 6
cut -d":" -f3-6 /etc/passwd  --> field 3 to 6

tr

echo "Shell Scripting" | tr "[a-z]" "[A-Z]"
echo "Shell Scripting" | tr "[:lower:]" "[:upper:]"
echo "Shell Scripting" | tr  "[A-Z]" "[a-z]"

Squeeze repeating occurrences of characters:

ps | tr -s " "
ps | tr -s "[:space:]"

Delete characters:

echo "My login pin is 5634" | tr -d "[:digit:]"
